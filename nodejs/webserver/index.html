创建 webSever

const http = require('http');

const server = http.createServer((req, res) => {
  if (req.url === '/ping') {
    res.end('pong');
  }
  res.end('hello web service');
});

server.listen(3000, () => {
  console.log('http start, listen on port 3000');
});
